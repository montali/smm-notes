\begin{example}
    The following matrix is symmetric
    $$
        \mat{A} = \left[\begin{matrix}
        1 & 2 & 3 \\
        2 & 1 & 4 \\
        3 & 4 & 1
        \end{matrix}\right]
    $$
    The following matrix is antisymmetric
    \renewcommand*{\arraystretch}{1.5}
    $$
        \mat{A} = \left[\begin{matrix}
        0 & 2 & -3 \\
        -2 & 0 & -4 \\
        3 & 4 & 0
        \end{matrix}\right]
    $$
    The following matrix is orthogonal
    $$
        \mat{A} = \left[\begin{matrix}
        0 & -1 & 0 \\
        1 & 0 & 0 \\
        0 & 0 & -1
        \end{matrix}\right]
    $$
\end{example}
